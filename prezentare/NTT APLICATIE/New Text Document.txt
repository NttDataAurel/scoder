CREATE TABLE `user_skill` (
	`ID` INT(5) NOT NULL,
	`USER_ID` INT(5) NOT NULL,
	`PHASE_ID` INT(5) NOT NULL,
	`NAME` VARCHAR(128) NOT NULL,
	`LEVEL` BIT(2) NOT NULL,
	`COMMENTS` TEXT NOT NULL,
	PRIMARY KEY (`ID`),
	INDEX `FK_user_skill_user` (`USER_ID`),
	CONSTRAINT `FK_user_skill_user` FOREIGN KEY (`USER_ID`) REFERENCES `user` (`USER_ID`)
);
